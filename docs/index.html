# Canon Kernel boot (single file)
from typing import Any, Dict, List, Tuple

class Scores:
    def __init__(self,k,t,s): self.kappa=k; self.tau=t; self.sigma=s
    def gate(self,thr): kmin,tmin,smax = thr; return (self.kappa>=kmin and self.tau>=tmin and self.sigma<=smax)

def zero_balance(v: List[float])->List[float]:
    m = sum(v)/len(v); return [x - m for x in v]

def fact_guard(text:str)->Tuple[str,float]:
    # toy confidence: mark likely-unsupported declaratives
    conf = 0.85 if any(w in text.lower() for w in ["source","cite","evidence"]) else 0.6
    if conf < 0.8: text += "\n[Grounding: hypotheses present; verification advised.]"
    return text, conf

class CanonKernel:
    thresholds = (0.65, 0.70, 0.25)  # κ_min, τ_min, Σ_max
    def phase_pi(self, prompt:str)->Dict[str,Any]:
        return {"prompt":prompt, "facets":prompt.split()}
    def phase_phi(self, obs:Dict[str,Any])->Dict[str,Any]:
        _ = zero_balance([1.0,-0.6,0.4])  # example balancing step
        return obs
    def phase_e(self, integ:Dict[str,Any])->str:
        p = integ["prompt"]
        return f"Plan for: {p}\n1) Perceive → 2) Integrate → 3) Act\n(Gated by κ/τ/Σ and fact-guard)."
    def score(self, draft:str, conf:float)->Scores:
        kappa = min(1.0, 0.75 + 0.20*conf)   # coherence
        tau   = 0.80                         # temporal responsibility (toy)
        sigma = 0.18                         # systemic risk (toy)
        return Scores(kappa,tau,sigma)
    def respond(self, prompt:str)->str:
        obs   = self.phase_pi(prompt)
        integ = self.phase_phi(obs)
        out   = self.phase_e(integ)
        out, conf = fact_guard(out)
        scores = self.score(out, conf)
        if not scores.gate(self.thresholds):
            out += "\n[Ethical gate: revise or gather more evidence.]"
        return out

# quick demo
if __name__ == "__main__":
    k = CanonKernel()
    print(k.respond("Design a rollout plan for a safe healthcare AI assistant"))
